-- ============================================================
-- CREACION DE BASES DE DATOS DISTRIBUIDAS - SISTEMA VETERINARIA
-- ============================================================
-- Simulamos 2 sucursales de veterinaria en diferentes ubicaciones
-- veterinaria_norte: Maneja informacion de clientes, mascotas y empleados
-- veterinaria_sur: Maneja servicios, citas, inventario y finanzas
-- ============================================================

-- Eliminar bases de datos si existen (para empezar limpio)
DROP DATABASE IF EXISTS veterinaria_norte;
DROP DATABASE IF EXISTS veterinaria_sur;

-- Crear base de datos de la Sucursal Norte
CREATE DATABASE veterinaria_norte;

-- Crear base de datos de la Sucursal Sur
CREATE DATABASE veterinaria_sur;

-- ============================================================
-- BASE DE DATOS: VETERINARIA_NORTE (Sucursal Norte)
-- Almacena: Clientes, Mascotas, Empleados, Proveedores
-- ============================================================

USE veterinaria_norte;

-- Tabla: DUENOS (Propietarios de las mascotas)
-- Almacena informacion completa de los clientes
CREATE TABLE duenos (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(100) NOT NULL,
    apellido VARCHAR(100) NOT NULL,
    telefono VARCHAR(15),
    telefono_emergencia VARCHAR(15),
    direccion VARCHAR(200),
    ciudad VARCHAR(50) DEFAULT 'Los Mochis',
    estado VARCHAR(50) DEFAULT 'Sinaloa',
    codigo_postal VARCHAR(10),
    email VARCHAR(100) UNIQUE,
    fecha_registro DATE DEFAULT (CURRENT_DATE),
    tipo_cliente ENUM('regular', 'premium', 'vip') DEFAULT 'regular',
    descuento_porcentaje DECIMAL(5,2) DEFAULT 0.00,
    saldo_favor DECIMAL(10,2) DEFAULT 0.00,
    observaciones TEXT,
    activo BOOLEAN DEFAULT TRUE,
    INDEX idx_nombre (nombre, apellido),
    INDEX idx_email (email),
    INDEX idx_tipo (tipo_cliente)
);

-- Tabla: MASCOTAS (Animales registrados)
-- Contiene informacion detallada de cada mascota
CREATE TABLE mascotas (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(50) NOT NULL,
    especie ENUM('Perro', 'Gato', 'Ave', 'Conejo', 'Hamster', 'Reptil', 'Otro') NOT NULL,
    raza VARCHAR(50),
    sexo ENUM('Macho', 'Hembra') NOT NULL,
    color VARCHAR(50),
    fecha_nacimiento DATE,
    edad INT,
    peso DECIMAL(6,2),
    altura DECIMAL(5,2),
    microchip VARCHAR(20) UNIQUE,
    esterilizado BOOLEAN DEFAULT FALSE,
    alergias TEXT,
    enfermedades_cronicas TEXT,
    seguro_medico VARCHAR(100),
    numero_poliza VARCHAR(50),
    foto_url VARCHAR(255),
    dueno_id INT NOT NULL,
    fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP,
    activo BOOLEAN DEFAULT TRUE,
    FOREIGN KEY (dueno_id) REFERENCES duenos(id) ON DELETE CASCADE,
    INDEX idx_dueno (dueno_id),
    INDEX idx_especie (especie),
    INDEX idx_microchip (microchip)
);

-- Tabla: EMPLEADOS (Personal de la veterinaria)
-- Registra doctores, enfermeros, recepcionistas, etc.
CREATE TABLE empleados (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(100) NOT NULL,
    apellido VARCHAR(100) NOT NULL,
    cedula_profesional VARCHAR(20) UNIQUE,
    especialidad VARCHAR(100),
    puesto ENUM('Veterinario', 'Enfermero', 'Recepcionista', 'Groomer', 'Administrador', 'Auxiliar') NOT NULL,
    telefono VARCHAR(15),
    email VARCHAR(100) UNIQUE,
    fecha_contratacion DATE NOT NULL,
    salario DECIMAL(10,2),
    horario_entrada TIME,
    horario_salida TIME,
    dias_laborales VARCHAR(50),
    activo BOOLEAN DEFAULT TRUE,
    INDEX idx_puesto (puesto),
    INDEX idx_cedula (cedula_profesional)
);

-- Tabla: PROVEEDORES (Empresas que suministran productos)
-- Almacena informacion de proveedores de medicamentos, alimentos, etc.
CREATE TABLE proveedores (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre_empresa VARCHAR(150) NOT NULL,
    contacto_nombre VARCHAR(100),
    contacto_telefono VARCHAR(15),
    contacto_email VARCHAR(100),
    direccion VARCHAR(200),
    ciudad VARCHAR(50),
    rfc VARCHAR(13) UNIQUE,
    tipo_productos VARCHAR(100),
    dias_credito INT DEFAULT 0,
    calificacion DECIMAL(3,2) DEFAULT 5.00,
    activo BOOLEAN DEFAULT TRUE,
    INDEX idx_nombre (nombre_empresa)
);

-- Tabla: RAZAS_COMUNES (Catalogo de razas por especie)
-- Facilita la seleccion de razas al registrar mascotas
CREATE TABLE razas_comunes (
    id INT PRIMARY KEY AUTO_INCREMENT,
    especie VARCHAR(30) NOT NULL,
    raza VARCHAR(50) NOT NULL,
    tamano ENUM('Toy', 'Pequeno', 'Mediano', 'Grande', 'Gigante') DEFAULT 'Mediano',
    temperamento VARCHAR(100),
    INDEX idx_especie (especie)
);

-- ============================================================
-- BASE DE DATOS: VETERINARIA_SUR (Sucursal Sur)
-- Almacena: Servicios, Citas, Historial Medico, Inventario, Finanzas
-- ============================================================

USE veterinaria_sur;

-- Tabla: SERVICIOS (Catalogo de servicios ofrecidos)
-- Lista todos los servicios disponibles con precios
CREATE TABLE servicios (
    id INT PRIMARY KEY AUTO_INCREMENT,
    codigo VARCHAR(20) UNIQUE NOT NULL,
    nombre VARCHAR(100) NOT NULL,
    descripcion TEXT,
    categoria ENUM('Consulta', 'Cirugia', 'Vacunacion', 'Estetica', 'Diagnostico', 'Emergencia', 'Hospitalizacion') NOT NULL,
    precio DECIMAL(8,2) NOT NULL,
    costo DECIMAL(8,2),
    duracion_minutos INT DEFAULT 30,
    requiere_cita BOOLEAN DEFAULT TRUE,
    activo BOOLEAN DEFAULT TRUE,
    INDEX idx_categoria (categoria),
    INDEX idx_codigo (codigo)
);

-- Tabla: CITAS (Agendamiento de servicios)
-- Registra todas las citas programadas y realizadas
CREATE TABLE citas (
    id INT PRIMARY KEY AUTO_INCREMENT,
    folio VARCHAR(20) UNIQUE NOT NULL,
    mascota_id INT NOT NULL,
    dueno_id INT NOT NULL,
    servicio_id INT NOT NULL,
    empleado_id INT NOT NULL,
    fecha_hora DATETIME NOT NULL,
    fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP,
    duracion_real INT,
    motivo_consulta TEXT,
    sintomas TEXT,
    temperatura DECIMAL(4,2),
    frecuencia_cardiaca INT,
    frecuencia_respiratoria INT,
    diagnostico TEXT,
    tratamiento TEXT,
    receta TEXT,
    estado ENUM('pendiente', 'en_proceso', 'completada', 'cancelada', 'no_asistio') DEFAULT 'pendiente',
    tipo_pago ENUM('efectivo', 'tarjeta', 'transferencia', 'seguro') DEFAULT 'efectivo',
    monto_total DECIMAL(10,2),
    monto_pagado DECIMAL(10,2) DEFAULT 0.00,
    observaciones TEXT,
    FOREIGN KEY (servicio_id) REFERENCES servicios(id),
    INDEX idx_mascota (mascota_id),
    INDEX idx_fecha (fecha_hora),
    INDEX idx_estado (estado),
    INDEX idx_folio (folio)
);

-- Tabla: HISTORIAL_MEDICO (Expediente clinico completo)
-- Almacena todo el historial medico de cada mascota
CREATE TABLE historial_medico (
    id INT PRIMARY KEY AUTO_INCREMENT,
    mascota_id INT NOT NULL,
    cita_id INT,
    fecha DATE NOT NULL,
    tipo_registro ENUM('Consulta', 'Cirugia', 'Vacuna', 'Analisis', 'Accidente', 'Seguimiento') NOT NULL,
    peso_registro DECIMAL(6,2),
    tratamiento TEXT,
    medicamentos TEXT,
    dosis VARCHAR(100),
    frecuencia VARCHAR(100),
    duracion_tratamiento VARCHAR(50),
    examenes_realizados TEXT,
    resultados_examenes TEXT,
    procedimientos TEXT,
    complicaciones TEXT,
    proximo_seguimiento DATE,
    observaciones TEXT,
    empleado_responsable INT,
    archivo_adjunto VARCHAR(255),
    FOREIGN KEY (cita_id) REFERENCES citas(id) ON DELETE SET NULL,
    INDEX idx_mascota (mascota_id),
    INDEX idx_fecha (fecha),
    INDEX idx_tipo (tipo_registro)
);

-- Tabla: VACUNAS (Control de vacunacion)
-- Registra todas las vacunas aplicadas
CREATE TABLE vacunas (
    id INT PRIMARY KEY AUTO_INCREMENT,
    mascota_id INT NOT NULL,
    nombre_vacuna VARCHAR(100) NOT NULL,
    laboratorio VARCHAR(100),
    lote VARCHAR(50),
    fecha_aplicacion DATE NOT NULL,
    fecha_vencimiento DATE,
    proxima_dosis DATE,
    empleado_aplica INT,
    reaccion_adversa TEXT,
    INDEX idx_mascota (mascota_id),
    INDEX idx_proxima (proxima_dosis)
);

-- Tabla: INVENTARIO (Control de productos)
-- Maneja el stock de medicamentos, alimentos, accesorios
CREATE TABLE inventario (
    id INT PRIMARY KEY AUTO_INCREMENT,
    codigo VARCHAR(20) UNIQUE NOT NULL,
    nombre VARCHAR(150) NOT NULL,
    descripcion TEXT,
    categoria ENUM('Medicamento', 'Alimento', 'Accesorio', 'Higiene', 'Instrumental', 'Consumible') NOT NULL,
    marca VARCHAR(50),
    presentacion VARCHAR(50),
    unidad_medida VARCHAR(20),
    stock_actual INT DEFAULT 0,
    stock_minimo INT DEFAULT 10,
    stock_maximo INT DEFAULT 100,
    precio_compra DECIMAL(10,2),
    precio_venta DECIMAL(10,2),
    proveedor_id INT,
    ubicacion_almacen VARCHAR(50),
    fecha_caducidad DATE,
    requiere_receta BOOLEAN DEFAULT FALSE,
    activo BOOLEAN DEFAULT TRUE,
    INDEX idx_codigo (codigo),
    INDEX idx_categoria (categoria),
    INDEX idx_stock (stock_actual)
);

-- Tabla: MOVIMIENTOS_INVENTARIO (Historial de movimientos)
-- Registra entradas, salidas y ajustes de inventario
CREATE TABLE movimientos_inventario (
    id INT PRIMARY KEY AUTO_INCREMENT,
    producto_id INT NOT NULL,
    tipo_movimiento ENUM('entrada', 'salida', 'ajuste', 'devolucion', 'merma') NOT NULL,
    cantidad INT NOT NULL,
    motivo VARCHAR(100),
    referencia VARCHAR(50),
    fecha_movimiento DATETIME DEFAULT CURRENT_TIMESTAMP,
    empleado_id INT,
    costo_unitario DECIMAL(10,2),
    observaciones TEXT,
    FOREIGN KEY (producto_id) REFERENCES inventario(id) ON DELETE CASCADE,
    INDEX idx_producto (producto_id),
    INDEX idx_fecha (fecha_movimiento),
    INDEX idx_tipo (tipo_movimiento)
);

-- Tabla: VENTAS (Registro de ventas de productos)
-- Almacena todas las transacciones de venta
CREATE TABLE ventas (
    id INT PRIMARY KEY AUTO_INCREMENT,
    folio VARCHAR(20) UNIQUE NOT NULL,
    dueno_id INT NOT NULL,
    fecha_venta DATETIME DEFAULT CURRENT_TIMESTAMP,
    subtotal DECIMAL(10,2) NOT NULL,
    descuento DECIMAL(10,2) DEFAULT 0.00,
    iva DECIMAL(10,2),
    total DECIMAL(10,2) NOT NULL,
    metodo_pago ENUM('efectivo', 'tarjeta', 'transferencia', 'credito') NOT NULL,
    estado ENUM('pagada', 'pendiente', 'cancelada') DEFAULT 'pagada',
    empleado_vendedor INT,
    observaciones TEXT,
    INDEX idx_fecha (fecha_venta),
    INDEX idx_dueno (dueno_id),
    INDEX idx_folio (folio)
);

-- Tabla: DETALLE_VENTAS (Productos vendidos por transaccion)
-- Detalla cada producto incluido en una venta
CREATE TABLE detalle_ventas (
    id INT PRIMARY KEY AUTO_INCREMENT,
    venta_id INT NOT NULL,
    producto_id INT NOT NULL,
    cantidad INT NOT NULL,
    precio_unitario DECIMAL(10,2) NOT NULL,
    descuento DECIMAL(10,2) DEFAULT 0.00,
    subtotal DECIMAL(10,2) NOT NULL,
    FOREIGN KEY (venta_id) REFERENCES ventas(id) ON DELETE CASCADE,
    FOREIGN KEY (producto_id) REFERENCES inventario(id),
    INDEX idx_venta (venta_id),
    INDEX idx_producto (producto_id)
);

-- Tabla: HOSPITALIZACION (Control de mascotas hospitalizadas)
-- Registra mascotas que requieren internamiento
CREATE TABLE hospitalizacion (
    id INT PRIMARY KEY AUTO_INCREMENT,
    mascota_id INT NOT NULL,
    fecha_ingreso DATETIME NOT NULL,
    fecha_salida DATETIME,
    motivo TEXT NOT NULL,
    jaula_numero VARCHAR(10),
    estado ENUM('activa', 'alta', 'referido', 'fallecido') DEFAULT 'activa',
    costo_diario DECIMAL(8,2),
    empleado_responsable INT,
    observaciones_ingreso TEXT,
    observaciones_salida TEXT,
    INDEX idx_mascota (mascota_id),
    INDEX idx_estado (estado)
);

-- Tabla: NOTAS_HOSPITALIZACION (Bitacora de hospitalizacion)
-- Registro diario del estado de mascotas hospitalizadas
CREATE TABLE notas_hospitalizacion (
    id INT PRIMARY KEY AUTO_INCREMENT,
    hospitalizacion_id INT NOT NULL,
    fecha_hora DATETIME DEFAULT CURRENT_TIMESTAMP,
    temperatura DECIMAL(4,2),
    frecuencia_cardiaca INT,
    frecuencia_respiratoria INT,
    alimento_consumido VARCHAR(100),
    medicamentos_administrados TEXT,
    comportamiento TEXT,
    observaciones TEXT,
    empleado_registro INT,
    FOREIGN KEY (hospitalizacion_id) REFERENCES hospitalizacion(id) ON DELETE CASCADE,
    INDEX idx_hospitalizacion (hospitalizacion_id),
    INDEX idx_fecha (fecha_hora)
);

-- Tabla: GASTOS (Control de gastos operativos)
-- Registra todos los gastos de la veterinaria
CREATE TABLE gastos (
    id INT PRIMARY KEY AUTO_INCREMENT,
    fecha DATE NOT NULL,
    categoria ENUM('Nomina', 'Servicios', 'Mantenimiento', 'Compras', 'Impuestos', 'Renta', 'Publicidad', 'Otros') NOT NULL,
    concepto VARCHAR(150) NOT NULL,
    monto DECIMAL(10,2) NOT NULL,
    metodo_pago ENUM('efectivo', 'tarjeta', 'transferencia', 'cheque') NOT NULL,
    proveedor_id INT,
    factura VARCHAR(50),
    comprobante_url VARCHAR(255),
    empleado_autoriza INT,
    observaciones TEXT,
    INDEX idx_fecha (fecha),
    INDEX idx_categoria (categoria)
);

-- CONSULTAS UTILES PARA VISUALIZAR LA BASE DE DATOS

-- Consulta 1: Ver todos los dueños con sus mascotas
-- Muestra la relación entre propietarios y sus animales
SELECT 
    d.id AS dueno_id,
    CONCAT(d.nombre, ' ', d.apellido) AS dueno,
    d.telefono,
    d.tipo_cliente,
    m.nombre AS mascota,
    m.especie,
    m.raza,
    m.edad,
    m.sexo
FROM veterinaria_norte.duenos d
LEFT JOIN veterinaria_norte.mascotas m ON d.id = m.dueno_id
WHERE d.activo = TRUE
ORDER BY d.apellido, m.nombre;

-- Consulta 2: Historial médico completo de una mascota
-- Permite ver todo el expediente clínico de un paciente
SELECT 
    m.nombre AS mascota,
    CONCAT(d.nombre, ' ', d.apellido) AS dueno,
    h.fecha,
    h.tipo_registro,
    h.peso_registro,
    h.tratamiento,
    h.medicamentos,
    h.diagnostico,
    h.observaciones
FROM veterinaria_norte.mascotas m
JOIN veterinaria_norte.duenos d ON m.dueno_id = d.id
LEFT JOIN veterinaria_sur.historial_medico h ON m.id = h.mascota_id
WHERE m.id = 1  -- Cambiar por el ID de la mascota deseada
ORDER BY h.fecha DESC;

-- Consulta 3: Citas pendientes y programadas
-- Lista todas las citas futuras o pendientes de atención
SELECT 
    c.folio,
    c.fecha_hora,
    m.nombre AS mascota,
    CONCAT(d.nombre, ' ', d.apellido) AS dueno,
    d.telefono,
    s.nombre AS servicio,
    CONCAT(e.nombre, ' ', e.apellido) AS veterinario,
    c.estado,
    c.monto_total
FROM veterinaria_sur.citas c
JOIN veterinaria_norte.mascotas m ON c.mascota_id = m.id
JOIN veterinaria_norte.duenos d ON c.dueno_id = d.id
JOIN veterinaria_sur.servicios s ON c.servicio_id = s.id
JOIN veterinaria_norte.empleados e ON c.empleado_id = e.id
WHERE c.estado = 'pendiente'
ORDER BY c.fecha_hora;

-- Consulta 4: Inventario con stock bajo
-- Identifica productos que requieren reorden
SELECT 
    i.codigo,
    i.nombre,
    i.categoria,
    i.stock_actual,
    i.stock_minimo,
    (i.stock_minimo - i.stock_actual) AS faltante,
    p.nombre_empresa AS proveedor,
    p.contacto_telefono
FROM veterinaria_sur.inventario i
LEFT JOIN veterinaria_norte.proveedores p ON i.proveedor_id = p.id
WHERE i.stock_actual <= i.stock_minimo
AND i.activo = TRUE
ORDER BY (i.stock_minimo - i.stock_actual) DESC;

-- Consulta 5: Reporte de ventas mensual
-- Muestra resumen de ingresos por ventas
SELECT 
    DATE_FORMAT(fecha_venta, '%Y-%m') AS mes,
    COUNT(*) AS total_ventas,
    SUM(subtotal) AS subtotal_mes,
    SUM(descuento) AS descuentos_mes,
    SUM(iva) AS iva_mes,
    SUM(total) AS total_mes,
    AVG(total) AS ticket_promedio
FROM veterinaria_sur.ventas
WHERE estado = 'pagada'
GROUP BY DATE_FORMAT(fecha_venta, '%Y-%m')
ORDER BY mes DESC;

-- Consulta 6: Productos más vendidos
-- Identifica los artículos con mayor rotación
SELECT 
    i.nombre AS producto,
    i.categoria,
    SUM(dv.cantidad) AS unidades_vendidas,
    SUM(dv.subtotal) AS ingresos_totales,
    COUNT(DISTINCT dv.venta_id) AS numero_ventas
FROM veterinaria_sur.detalle_ventas dv
JOIN veterinaria_sur.inventario i ON dv.producto_id = i.id
GROUP BY i.id, i.nombre, i.categoria
ORDER BY unidades_vendidas DESC
LIMIT 20;

-- Consulta 7: Próximas vacunas por aplicar
-- Lista mascotas que requieren refuerzo de vacunación
SELECT 
    m.nombre AS mascota,
    CONCAT(d.nombre, ' ', d.apellido) AS dueno,
    d.telefono,
    v.nombre_vacuna,
    v.proxima_dosis
FROM veterinaria_sur.vacunas v
JOIN veterinaria_norte.mascotas m ON v.mascota_id = m.id
JOIN veterinaria_norte.duenos d ON m.dueno_id = d.id
WHERE v.proxima_dosis IS NOT NULL
AND v.proxima_dosis BETWEEN CURDATE() AND DATE_ADD(CURDATE(), INTERVAL 30 DAY)
ORDER BY v.proxima_dosis;

-- Consulta 8: Mascotas hospitalizadas actualmente
-- Muestra pacientes en internamiento
SELECT 
    m.nombre AS mascota,
    m.especie,
    CONCAT(d.nombre, ' ', d.apellido) AS dueno,
    d.telefono,
    h.motivo,
    h.jaula_numero,
    h.fecha_ingreso,
    DATEDIFF(NOW(), h.fecha_ingreso) AS dias_hospitalizacion,
    CONCAT(e.nombre, ' ', e.apellido) AS veterinario_responsable
FROM veterinaria_sur.hospitalizacion h
JOIN veterinaria_norte.mascotas m ON h.mascota_id = m.id
JOIN veterinaria_norte.duenos d ON m.dueno_id = d.id
JOIN veterinaria_norte.empleados e ON h.empleado_responsable = e.id
WHERE h.estado = 'activa'
ORDER BY h.fecha_ingreso;

-- Consulta 9: Reporte financiero consolidado mensual
-- Vista completa de ingresos y egresos
SELECT 
    'Ingresos Consultas' AS concepto,
    SUM(monto_total) AS monto,
    DATE_FORMAT(fecha_hora, '%Y-%m') AS mes
FROM veterinaria_sur.citas
WHERE estado = 'completada'
GROUP BY DATE_FORMAT(fecha_hora, '%Y-%m')

UNION ALL

SELECT 
    'Ingresos Ventas' AS concepto,
    SUM(total) AS monto,
    DATE_FORMAT(fecha_venta, '%Y-%m') AS mes
FROM veterinaria_sur.ventas
WHERE estado = 'pagada'
GROUP BY DATE_FORMAT(fecha_venta, '%Y-%m')

UNION ALL

SELECT 
    CONCAT('Gasto: ', categoria) AS concepto,
    -SUM(monto) AS monto,
    DATE_FORMAT(fecha, '%Y-%m') AS mes
FROM veterinaria_sur.gastos
GROUP BY categoria, DATE_FORMAT(fecha, '%Y-%m')

ORDER BY mes DESC, monto DESC;

-- Consulta 10: Clientes VIP y Premium con mayor gasto
-- Identifica los mejores clientes
SELECT 
    CONCAT(d.nombre, ' ', d.apellido) AS cliente,
    d.tipo_cliente,
    d.telefono,
    d.email,
    COUNT(DISTINCT m.id) AS total_mascotas,
    COUNT(DISTINCT c.id) AS total_citas,
    COALESCE(SUM(c.monto_total), 0) AS gasto_consultas,
    COALESCE(SUM(v.total), 0) AS gasto_ventas,
    COALESCE(SUM(c.monto_total), 0) + COALESCE(SUM(v.total), 0) AS gasto_total
FROM veterinaria_norte.duenos d
LEFT JOIN veterinaria_norte.mascotas m ON d.id = m.dueno_id
LEFT JOIN veterinaria_sur.citas c ON d.id = c.dueno_id AND c.estado = 'completada'
LEFT JOIN veterinaria_sur.ventas v ON d.id = v.dueno_id AND v.estado = 'pagada'
WHERE d.tipo_cliente IN ('premium', 'vip')
AND d.activo = TRUE
GROUP BY d.id, d.nombre, d.apellido, d.tipo_cliente, d.telefono, d.email
ORDER BY gasto_total DESC
LIMIT 20;

-- ============================================================
-- CONSULTAS AVANZADAS PARA ANÁLISIS DISTRIBUIDO
-- ============================================================

-- Consulta 11: Vista federada - Información completa del paciente
-- Combina datos de ambas bases de datos distribuidas
SELECT 
    -- Información del Norte (Clientes y Mascotas)
    m.id AS mascota_id,
    m.nombre AS mascota,
    m.especie,
    m.raza,
    m.edad,
    m.microchip,
    CONCAT(d.nombre, ' ', d.apellido) AS propietario,
    d.telefono,
    d.ciudad,
    d.tipo_cliente,
    
    -- Información del Sur (Servicios Médicos)
    COUNT(DISTINCT c.id) AS total_consultas,
    COUNT(DISTINCT v.id) AS total_vacunas,
    COUNT(DISTINCT h.id) AS registros_historial,
    MAX(c.fecha_hora) AS ultima_consulta,
    MAX(v.fecha_aplicacion) AS ultima_vacuna
    
FROM veterinaria_norte.mascotas m
JOIN veterinaria_norte.duenos d ON m.dueno_id = d.id
LEFT JOIN veterinaria_sur.citas c ON m.id = c.mascota_id
LEFT JOIN veterinaria_sur.vacunas v ON m.id = v.mascota_id
LEFT JOIN veterinaria_sur.historial_medico h ON m.id = h.mascota_id
WHERE m.activo = TRUE
GROUP BY m.id, m.nombre, m.especie, m.raza, m.edad, m.microchip,
         d.nombre, d.apellido, d.telefono, d.ciudad, d.tipo_cliente
ORDER BY total_consultas DESC;
